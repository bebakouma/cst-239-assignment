001package edu.gcu.storefront;
002
003import java.math.BigDecimal;
004
005/**
006 * Common behaviour for anything that can be sold in the store.
007 */
008public interface Salable {
009
010    /** Unique identifier (SKU, UUID, etc.). */
011    String getId();
012
013    /** Human-readable name. */
014    String getName();
015
016    /** Longer description shown to customers. */
017    String getDescription();
018
019    /** Unit price expressed as {@link BigDecimal} to avoid floating-point errors. */
020    BigDecimal getPrice();
021
022    /** Current on-hand quantity. */
023    int getQuantity();
024
025    /**
026     * Adjust inventory.
027     *
028     * @param delta positive to add stock; negative to deduct.
029     * @throws IllegalArgumentException if the adjustment would take inventory below zero.
030     */
031    void adjustQuantity(int delta);
032}
